justice.js

How to use

      <script type="text/javascript" src="justice.min.js"></script>
      <script type="text/javascript">
        Justice.init();
      </script>

How to use, with options

      <script type="text/javascript" src="justice.min.js"></script>
      <script type="text/javascript">
        Justice.init({
          metrics: {
            pageLoad: { budget: 800 },
            domComplete: { budget: 600 },
            domInteractive: { budget: 400 },
            requests: { budget: 6 }
          },
          showFPS: true,
          chartType: 'spline'
        });
      </script>
